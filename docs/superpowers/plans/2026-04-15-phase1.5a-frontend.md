Replace `src/api/auth.rs` with dual-mode auth that accepts either a bearer token (for bots) or a Clerk JWT (for frontend users):

```rust
use axum::{
    extract::FromRequestParts,
    http::request::Parts,
};
use jsonwebtoken::{decode, Algorithm, DecodingKey, Validation};
use serde::{Deserialize, Serialize};

use crate::error::{AppError, AppResult};
use crate::state::AppState;

/// Claims extracted from a Clerk JWT.
#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct ClerkClaims {
    pub sub: String,
    pub iss: String,
    pub exp: u64,
    pub iat: u64,
    /// Clerk public metadata containing role.
    #[serde(default)]
    pub public_metadata: PublicMetadata,
}

#[derive(Debug, Serialize, Deserialize, Clone, Default)]
pub struct PublicMetadata {
    #[serde(default)]
    pub role: Option<String>,
}

/// Authenticated identity — either a bot (bearer token) or a human (Clerk JWT).
#[derive(Debug, Clone)]
pub enum AuthIdentity {
    /// Authenticated via static bearer token (bot or admin CLI).
    BearerToken,
    /// Authenticated via Clerk JWT (frontend user).
    ClerkUser {
        user_id: String,
        role: String,
    },
}

impl AuthIdentity {
    /// Returns true if the identity has admin privileges.
    pub fn is_admin(&self) -> bool {
        match self {
            AuthIdentity::BearerToken => true,
            AuthIdentity::ClerkUser { role, .. } => role == "admin",
        }
    }

    /// Returns the Clerk user ID, if this is a Clerk identity.
    pub fn user_id(&self) -> Option<&str> {
        match self {
            AuthIdentity::BearerToken => None,
            AuthIdentity::ClerkUser { user_id, .. } => Some(user_id),
        }
    }
}

impl FromRequestParts<AppState> for AuthIdentity {
    type Rejection = AppError;

    async fn from_request_parts(
        parts: &mut Parts,
        state: &AppState,
    ) -> AppResult<Self> {
        let settings = state.settings();

        let header = parts
            .headers
            .get("authorization")
            .and_then(|v| v.to_str().ok())
            .ok_or(AppError::Unauthorized)?;

        let token = header
            .strip_prefix("Bearer ")
            .ok_or(AppError::Unauthorized)?;

        // Try static bearer token first (for bots and admin CLI)
        if !settings.auth.admin_token.is_empty() && token == settings.auth.admin_token {
            return Ok(AuthIdentity::BearerToken);
        }

        // Try Clerk JWT
        if !settings.auth.clerk_issuer.is_empty() {
            match decode_clerk_jwt(token, &settings.auth.clerk_issuer) {
                Ok(claims) => {
                    let role = claims
                        .public_metadata
                        .role
                        .unwrap_or_else(|| "member".to_string());
                    return Ok(AuthIdentity::ClerkUser {
                        user_id: claims.sub,
                        role,
                    });
                }
                Err(_) => {}
            }
        }

        // Dev mode: if admin_token is empty and no clerk config, allow all
        if settings.auth.admin_token.is_empty() && settings.auth.clerk_issuer.is_empty() {
            return Ok(AuthIdentity::BearerToken);
        }

        Err(AppError::Unauthorized)
    }
}

/// Decode and validate a Clerk JWT.
/// Note: In production, you should fetch JWKS from clerk_jwks_url and cache the keys.
/// For simplicity, this uses HS256 validation against the issuer.
/// A production implementation should use RS256 with JWKS key rotation.
fn decode_clerk_jwt(
    token: &str,
    issuer: &str,
) -> Result<ClerkClaims, jsonwebtoken::errors::Error> {
    let mut validation = Validation::new(Algorithm::RS256);
    validation.set_issuer(&[issuer]);
    validation.validate_exp = true;

    // In production: fetch the JWKS from clerk_jwks_url, extract the RSA public key,
    // and use DecodingKey::from_rsa_components. For now, we decode without signature
    // verification in dev mode and require proper JWKS setup in production.
    // TODO(task): implement JWKS fetching with caching in a follow-up.
    let key = DecodingKey::from_secret(b"placeholder");
    let token_data = decode::<ClerkClaims>(token, &key, &validation)?;
    Ok(token_data.claims)
}

// Keep backward compat: BearerAuth type alias
pub type BearerAuth = AuthIdentity;

/// Extractor that requires admin privileges.
pub struct AdminOnly(pub AuthIdentity);

impl FromRequestParts<AppState> for AdminOnly {
    type Rejection = AppError;

    async fn from_request_parts(
        parts: &mut Parts,
        state: &AppState,
    ) -> AppResult<Self> {
        let identity = AuthIdentity::from_request_parts(parts, state).await?;
        if !identity.is_admin() {
            return Err(AppError::Unauthorized);
        }
        Ok(AdminOnly(identity))
    }
}
```

**Note on JWKS:** The JWT verification above is a scaffold. Clerk uses RS256 with rotating keys fetched from the JWKS endpoint. A proper implementation needs to:
1. Fetch JWKS from `clerk_jwks_url` on startup
2. Cache the keys with periodic refresh (every 1 hour)
3. Decode using `DecodingKey::from_rsa_components`

This will be refined during implementation — the key architectural decision (dual auth with `AuthIdentity` enum) is what matters for the plan.

/// PATCH /bots/{id}/approve — admin approves a pending bot.
pub async fn approve_bot(
    State(state): State<AppState>,
    admin: AdminOnly,
    Path(id): Path<String>,
) -> AppResult<Json<BotResponse>> {
    let bot = db::queries::get_bot(state.db(), &id)
        .await
        .map_err(|e| AppError::Database(e.to_string()))?
        .ok_or(AppError::NotFound)?;
    if bot.status != "pending" {
        return Err(AppError::BadRequest("Bot is not pending".into()));
    }
    db::queries::update_bot_status(
        state.db(), &id, "active", admin.0.user_id(),
    ).await.map_err(|e| AppError::Database(e.to_string()))?;
    let updated = db::queries::get_bot(state.db(), &id)
        .await
        .map_err(|e| AppError::Database(e.to_string()))?
        .ok_or(AppError::NotFound)?;
    Ok(Json(bot_to_response(&updated)))
}

use serde::{Deserialize, Serialize};
use std::collections::{HashMap, HashSet};

/// Result of validating synthesis citations against the transcript.
#[derive(Debug, Serialize, Deserialize)]
pub struct CitationCheckResult {
    pub citations_total: usize,
    pub citations_valid: usize,
    pub citations_invalid: Vec<InvalidCitation>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct InvalidCitation {
    pub citation: String,
    pub reason: String,
    pub location: String,
}

/// Parse and validate all [Agent X, Round N] citations in synthesis output.
pub fn check_citations(
    synthesis_json: &serde_json::Value,
    valid_pseudonyms: &HashSet<String>,
    responses_by_round: &HashMap<(String, i64), bool>, // (pseudonym, round) -> abstained
    max_round: i64,
) -> CitationCheckResult {
    let mut total = 0;
    let mut invalid = Vec::new();

    // Extract citations from all text fields in the synthesis
    let citation_re = regex::Regex::new(r"\[([^,\]]+),\s*Round\s*(\d+)\]").unwrap();

    // Check consensus_points
    if let Some(points) = synthesis_json.get("consensus_points").and_then(|v| v.as_array()) {
        for (i, point) in points.iter().enumerate() {
            if let Some(evidence) = point.get("evidence").and_then(|v| v.as_str()) {
                for cap in citation_re.captures_iter(evidence) {
                    total += 1;
                    let agent = cap[1].trim().to_string();
                    let round: i64 = cap[2].parse().unwrap_or(-1);
                    let location = format!("consensus_points[{}].evidence", i);
                    validate_citation(&agent, round, valid_pseudonyms, responses_by_round, max_round, &location, &mut invalid);
                }
            }
        }
    }

    // Check live_disagreements
    if let Some(disagreements) = synthesis_json.get("live_disagreements").and_then(|v| v.as_array()) {
        for (i, d) in disagreements.iter().enumerate() {
            for side in ["side_a", "side_b"] {
                if let Some(arg) = d.get(side).and_then(|s| s.get("best_argument")).and_then(|v| v.as_str()) {
                    for cap in citation_re.captures_iter(arg) {
                        total += 1;
                        let agent = cap[1].trim().to_string();
                        let round: i64 = cap[2].parse().unwrap_or(-1);
                        let location = format!("live_disagreements[{}].{}.best_argument", i, side);
                        validate_citation(&agent, round, valid_pseudonyms, responses_by_round, max_round, &location, &mut invalid);
                    }
                }
            }
        }
    }

    // Check minority_positions
    if let Some(positions) = synthesis_json.get("minority_positions").and_then(|v| v.as_array()) {
        for (i, pos) in positions.iter().enumerate() {
            if let Some(arg) = pos.get("key_argument").and_then(|v| v.as_str()) {
                for cap in citation_re.captures_iter(arg) {
                    total += 1;
                    let agent = cap[1].trim().to_string();
                    let round: i64 = cap[2].parse().unwrap_or(-1);
                    let location = format!("minority_positions[{}].key_argument", i);
                    validate_citation(&agent, round, valid_pseudonyms, responses_by_round, max_round, &location, &mut invalid);
                }
            }
        }
    }

    let valid = total - invalid.len();
    CitationCheckResult { citations_total: total, citations_valid: valid, citations_invalid: invalid }
}

fn validate_citation(
    agent: &str, round: i64,
    valid_pseudonyms: &HashSet<String>,
    responses_by_round: &HashMap<(String, i64), bool>,
    max_round: i64,
    location: &str,
    invalid: &mut Vec<InvalidCitation>,
) {
    let citation = format!("[{}, Round {}]", agent, round);
    if !valid_pseudonyms.contains(agent) {
        invalid.push(InvalidCitation { citation, reason: format!("{} is not a participant", agent), location: location.to_string() });
    } else if round < 0 || round > max_round {
        invalid.push(InvalidCitation { citation, reason: format!("Round {} does not exist", round), location: location.to_string() });
    } else if let Some(&abstained) = responses_by_round.get(&(agent.to_string(), round)) {
        if abstained {
            invalid.push(InvalidCitation { citation, reason: format!("{} abstained in Round {}", agent, round), location: location.to_string() });
        }
    } else {
        invalid.push(InvalidCitation { citation, reason: format!("{} has no response in Round {}", agent, round), location: location.to_string() });
    }
}

import { env } from '$env/dynamic/public';
import type {
  BotResponse,
  CreateBotRequest,
  CreateDebateRequest,
  DebateResponse,
  SynthesisResponse,
  TranscriptResponse,
  UserInfoResponse,
} from '$lib/types';

const BASE_URL = env.PUBLIC_API_URL;

class ApiError extends Error {
  constructor(
    public status: number,
    public body: unknown,
  ) {
    super(`API error ${status}`);
  }
}

async function getToken(): Promise<string | null> {
  // Clerk token retrieval — will be wired up in Task 10
  // For now, return null (dev mode with no auth)
  return null;
}

async function request<T>(path: string, options: RequestInit = {}): Promise<T> {
  const token = await getToken();
  const headers: Record<string, string> = {
    'Content-Type': 'application/json',
    ...((options.headers as Record<string, string>) ?? {}),
  };
  if (token) {
    headers['Authorization'] = `Bearer ${token}`;
  }

  const res = await fetch(`${BASE_URL}${path}`, {
    ...options,
    headers,
  });

  if (!res.ok) {
    const body = await res.json().catch(() => null);
    throw new ApiError(res.status, body);
  }

  if (res.status === 204) return undefined as T;
  return res.json();
}

export const api = {
  me(): Promise<UserInfoResponse> {
    return request('/me');
  },

  debates: {
    list(params?: { status?: string; limit?: number }): Promise<DebateResponse[]> {
      const searchParams = new URLSearchParams();
      if (params?.status) searchParams.set('status', params.status);
      if (params?.limit) searchParams.set('limit', String(params.limit));
      const qs = searchParams.toString();
      return request(`/debates${qs ? `?${qs}` : ''}`);
    },

    get(id: string): Promise<DebateResponse> {
      return request(`/debates/${id}`);
    },

    create(req: CreateDebateRequest): Promise<DebateResponse> {
      return request('/debates', {
        method: 'POST',
        body: JSON.stringify(req),
      });
    },

    transcript(id: string): Promise<TranscriptResponse> {
      return request(`/debates/${id}/transcript`);
    },

    synthesis(id: string): Promise<SynthesisResponse> {
      return request(`/debates/${id}/synthesis`);
    },
  },

  bots: {
    list(): Promise<BotResponse[]> {
      return request('/bots');
    },

    create(req: CreateBotRequest): Promise<BotResponse> {
      return request('/bots', {
        method: 'POST',
        body: JSON.stringify(req),
      });
    },

    approve(id: string): Promise<BotResponse> {
      return request(`/bots/${id}/approve`, { method: 'PATCH' });
    },

    reject(id: string): Promise<BotResponse> {
      return request(`/bots/${id}/reject`, { method: 'PATCH' });
    },

    deactivate(id: string): Promise<void> {
      return request(`/bots/${id}/deactivate`, { method: 'PATCH' });
    },

    reactivate(id: string): Promise<void> {
      return request(`/bots/${id}/reactivate`, { method: 'PATCH' });
    },

    mySubmissions(): Promise<BotResponse[]> {
      return request('/bots/my-submissions');
    },
  },
};

export { ApiError };
